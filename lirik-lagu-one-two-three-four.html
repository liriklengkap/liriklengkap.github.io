[tex] \tt = 5 + ( \frac{5 \: \times \: 2}{5} ) + 100[/tex]

[tex] \tt = 5 + \frac{ \cancel10}{ \cancel5} + 100[/tex]

[tex] \tt = 7 + 100[/tex]
